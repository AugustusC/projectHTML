variable
  c' ℓ : Level

module withField (F : Field c' ℓ) where
  open Field F renaming (refl to reflₚ)

  fSemiring : Semiring c' ℓ
  fSemiring = record
                { Carrier = Carrier
                ; _≈_ = _≈_
                ; _+_ = _+_
                ; _*_ = _*_
                ; 0# = 0#
                ; 1# = 1#
                ; isSemiring = isSemiring
                }

  open import matrix
  
  open matrix.WithSemiring fSemiring

  diff : Carrier → Carrier → ℕ
  diff a b = if a ≈ᵇ b then 0 else 1

  ... | true = 0

  diff-cong = cong (λ b → if b then 0 else 1)
  
  distᵥ : Vector Carrier n → Vector Carrier n → ℕ
  distᵥ {zero} u v = 0
  distᵥ {suc n} u v = (diff (u zerof) (v zerof)) ℕ.+ distᵥ (u ∘ sucf) (v ∘ sucf)

  dist : Matrix Carrier m n → Matrix Carrier m n → ℕ
  dist {zero} {n} M N = 0
  dist {suc m} {n} M N = distᵥ (M zerof) (N zerof) ℕ.+ dist (M ∘ sucf) (N ∘ sucf)

  weight : Matrix Carrier m n → ℕ
  weight M = dist M 0ₘ

--d(u,v)=0 ⟺ u=v (Identity of Indiscernables)

  diffIdIndₗ : ∀ a b → diff a b ≡ 0 → a ≈ b
  diffIdIndₗ a b p = ≈ᵇ≈ a b (p₁ a b p) where
    p₀ : ∀ b → (if b then 0 else 1) ≡ 0 → b ≡ true
    p₀ true refl = refl
    p₁ : ∀ a b → diff a b ≡ 0 → a ≈ᵇ b ≡ true
    p₁ a b p = p₀ (a ≈ᵇ b) p
  
  diffIdIndᵣ : ∀ a b → a ≈ b → diff a b ≡ 0
  diffIdIndᵣ a b p = diff-cong (≈≈ᵇ a b p)

  idIndᵣᵥ : {n : ℕ} → (u v : Vector Carrier n) → (u ≈ᵥ v) → distᵥ u v ≡ 0
  idIndᵣᵥ {ℕ.zero} u v p = refl
  idIndᵣᵥ {suc n} u v p = cong₂ ℕ._+_ (diffIdIndᵣ _ _ (p _)) (idIndᵣᵥ (λ k → u (sucf k)) _ (λ i → p (sucf i)))
  
  idIndᵣ : {m n : ℕ} → (M N : Matrix Carrier m n) → M ≈ₘ N → dist M N ≡ 0
  idIndᵣ {ℕ.zero} {n} M N p = refl
  idIndᵣ {suc m} {n} M N p = cong₂ ℕ._+_ (idIndᵣᵥ (M zerof) _ (p _)) (idIndᵣ (λ k → M (sucf k)) _ (λ i → p (sucf i)))

  eqZeroₗ : ∀ a b → a ℕ.+ b ≡ 0 → a ≡ 0
  eqZeroₗ zero zero p = refl

  eqZeroᵣ : ∀ a b → a ℕ.+ b ≡ 0 → b ≡ 0
  eqZeroᵣ zero zero p = refl

  idIndₗᵥ : {n : ℕ} → (u v : Vector Carrier n) → distᵥ u v ≡ 0 → u ≈ᵥ v
  idIndₗᵥ u v p i = diffIdIndₗ (u i) _ (diffDist u v p i) where
    diffDist : {n : ℕ} → (u v : Vector Carrier n) → distᵥ u v ≡ 0 → (i : Fin n) → diff (u i) (v i) ≡ 0
    diffDist {suc n} u v p zerof = eqZeroₗ _ _ p
    diffDist {suc n} u v p (sucf i) = diffDist _ _ (eqZeroᵣ _ _ p) i

  idIndₗ : {m n : ℕ} → (M N : Matrix Carrier m n) → dist M N ≡ 0 → M ≈ₘ N
  idIndₗ {suc m} {n} M N p zerof j = idIndₗᵥ (M zerof) (N zerof) (eqZeroₗ _ _ p) j
  idIndₗ {suc m} {n} M N p (sucf i) j = idIndₗ _ _ (eqZeroᵣ _ _ p) i j
  
--d(u,v) = d(v,u) (Symmetry)    

  ≉sym : ∀ a b → ¬(a ≈ b) → ¬(b ≈ a)
  ≉sym a b x x₁ = ⊥-elim (x (sym x₁))

  ≈ᵇSym : ∀ a b → a ≈ᵇ b ≡ b ≈ᵇ a
  ≈ᵇSym a b with a ≈ᵇ b | inspect (a ≈ᵇ_) b
  ... | false | [ eq ] = ≡sym (¬≈≈ᵇ b a (≉sym a b (¬≈ᵇ≈ a b eq)))
  ... | true  | [ eq ] = ≡sym (≈≈ᵇ b a (sym (≈ᵇ≈ a b eq)))

  distSymᵥ : {n : ℕ} → (u v : Vector Carrier n) → distᵥ u v ≡ distᵥ v u
  distSymᵥ {ℕ.zero} u v = refl
  distSymᵥ {suc n} u v = cong₂ ℕ._+_ (diff-cong (≈ᵇSym (u zerof) _)) (distSymᵥ (λ k → u (sucf k)) _)

  distSym : {m n : ℕ} → (M N : Matrix Carrier m n) → dist M N ≡ dist N M
  distSym {ℕ.zero} {n} M N = refl
  distSym {suc m} {n} M N = cong₂ ℕ._+_  (distSymᵥ (M zerof) _) (distSym (λ k → M (sucf k)) _)

--d(u,v) ≤ d(u,w) + d(w,v) (Triangle Inequality)

  diff<2 : ∀ b n → (if b then 0 else 1) ≡ suc(suc n) → ⊥
  diff<2 false n ()
  diff<2 true n ()

  pdiff1 : ∀ a b → diff a b ≡ 1 → ¬(a ≈ b)
  pdiff1 a b p x = ¬≈ᵇ≈ a b p₁ x where
    p₀ : ∀ b → (if b then 0 else 1) ≡ 1 → b ≡ false
    p₀ false p = refl
    p₁ : a ≈ᵇ b ≡ false
    p₁ = p₀ (a ≈ᵇ b) p

  ptri : ∀ x y z → diff x y ≡ 1 → 1 ℕ.≤ diff x z ℕ.+ diff z y
  ptri x y z p with diff x z | inspect (diff x) z | diff z y | inspect (diff z) y
  ... | ℕ.zero | [ eq ] | ℕ.zero | [ eq₁ ] = ⊥-elim (pdiff1 x y p (trans (diffIdIndₗ x z eq) (diffIdIndₗ z y eq₁)))
  ... | ℕ.zero | [ eq ] | suc r' | [ eq₁ ] = s≤s z≤n
  ... | suc r  | [ eq ] | r'     | [ eq₁ ] = s≤s z≤n
  
  difftri : ∀ x y z → diff x y ℕ.≤ diff x z ℕ.+ diff z y
  difftri x y z with diff x y | inspect (diff x) y
  ... | 0           | [ eq ] = ℕ.z≤n
  ... | 1           | [ eq ] = ptri x y z eq
  ... | suc (suc n) | [ eq ] = ⊥-elim (diff<2 (x ≈ᵇ y) n eq)


  ≤+congʳ : ∀ a b c → a ℕ.≤ b → c ℕ.+ a ℕ.≤ c ℕ.+ b
  ≤+congʳ a b zero p = p
  ≤+congʳ a b (suc c) p = s≤s (≤+congʳ a b c p)

  ≤cong : ∀ a b c d → a ℕ.≤ b → a ≡ c → b ≡ d → c ℕ.≤ d
  ≤cong a b .a .b p refl refl = p

  ≤+congˡ : ∀ a b c → a ℕ.≤ b → a ℕ.+ c ℕ.≤ b ℕ.+ c
  ≤+congˡ a b c p = ≤cong (c ℕ.+ a ) _ _ _ (≤+congʳ a _ _ p) (ℕₚ.+-comm c _) (ℕₚ.+-comm c _)

  distTriᵥ : {n : ℕ} → (u v w : Vector Carrier n) → distᵥ u v ℕ.≤ (distᵥ u w) ℕ.+ (distᵥ w v)
  distTriᵥ {ℕ.zero} u v w = z≤n
  distTriᵥ {suc n} u v w = ≤-trans p₂ (≤-reflexive (p₃ (diff (u zerof) _) _ _ _)) where
    p₀ : distᵥ u v ℕ.≤ diff (u zerof) (v zerof) ℕ.+ distᵥ (u ∘ sucf) (v ∘ sucf)
    p₀ = ≤-reflexive refl
    p₁ : distᵥ u v
     ℕ.≤ diff (u zerof) (w zerof) ℕ.+ diff (w zerof) (v zerof) ℕ.+ distᵥ (u ∘ sucf) (v ∘ sucf)
    p₁ = ≤-trans p₀ (≤+congˡ (diff (u zerof) (v zerof)) _ _ (difftri _ _ _))
    p₂ : distᵥ u v
     ℕ.≤ (diff (u zerof) (w zerof) ℕ.+ diff (w zerof) (v zerof))
     ℕ.+ (distᵥ (u ∘ sucf) (w ∘ sucf) ℕ.+ distᵥ (w ∘ sucf) (v ∘ sucf))
    p₂ = ≤-trans p₁ (≤+congʳ _ _ _ (distTriᵥ (u ∘ sucf) _ _))
    p₃ : ∀ a b c d → (a ℕ.+ b) ℕ.+ (c ℕ.+ d) ≡ (a ℕ.+ c) ℕ.+ (b ℕ.+ d)
    p₃ a b c d =
      begin
        (a ℕ.+ b) ℕ.+ (c ℕ.+ d) ≡⟨ ≡sym (ℕₚ.+-assoc (a ℕ+ b) c d) ⟩
        ((a ℕ.+ b) ℕ.+ c) ℕ.+ d ≡⟨ cong₂ ℕ._+_ (cong₂ ℕ._+_ (ℕₚ.+-comm a b) refl) refl ⟩
        ((b ℕ.+ a) ℕ.+ c) ℕ.+ d ≡⟨ cong₂ ℕ._+_ (ℕₚ.+-assoc b a c) refl ⟩
        (b ℕ.+ (a ℕ.+ c)) ℕ.+ d ≡⟨ cong₂ ℕ._+_ (ℕₚ.+-comm b (a ℕ.+ c)) refl ⟩
        ((a ℕ.+ c) ℕ.+ b) ℕ.+ d ≡⟨ ℕₚ.+-assoc (a ℕ.+ c) b d ⟩
        (a ℕ.+ c) ℕ.+ (b ℕ.+ d)
      ∎
