record IsField (_≈ᵇ_ : A → A → 𝔹) (_+_ _*_ : Op₂ A) (-_ _⁻¹ : Op₁ A) (0# 1# : A) : Set (a ⊔ ℓ) where
  field
    isCommutativeRing : IsCommutativeRing _≈_ _+_ _*_ -_ 0# 1#
    ⁻¹inverse* : Inverse-¬0 _≈_ 0# 1# _⁻¹ _*_
    ≈ᵇ≈ : ≈ᵇ≡ _≈_ _≈ᵇ_
    ¬≈ᵇ≈ : ¬≈ᵇ≡ _≈_ _≈ᵇ_
    ≈≈ᵇ : ≡≈ᵇ _≈_ _≈ᵇ_
    ¬≈≈ᵇ : ¬≡≈ᵇ _≈_ _≈ᵇ_
    
  open IsCommutativeRing isCommutativeRing public
