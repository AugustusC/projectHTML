module inv0
  {a ℓ} {A : Set a}
  (_≈_ : Rel A ℓ)
  where

-- Note: This doesn't enforce that the value given to represent 0# is actually Zero
-- In the context of using this to define a field this shouldn't matter as isCommutativeRing will
-- ensure that the value given for 0# is Zero


LeftInverse-¬0 : A → A → Op₁ A → Op₂ A → Set _
LeftInverse-¬0 0# e _⁻¹ _∙_ = ∀ x → ¬(x ≈ 0#) → ((x ⁻¹) ∙ x) ≈ e

RightInverse-¬0 : A → A → Op₁ A → Op₂ A → Set _
RightInverse-¬0 0# e _⁻¹ _∙_ = ∀ x → ¬(x ≈ 0#) → (x ∙ (x ⁻¹)) ≈ e

Inverse-¬0 : A → A → Op₁ A → Op₂ A → Set _
Inverse-¬0 0# e _⁻¹ _∙_ =
  (LeftInverse-¬0 0# e _⁻¹ _∙_) × (RightInverse-¬0 0# e _⁻¹ _∙_)
