data 𝔽₂ : Set where
  zero : 𝔽₂
  one  : 𝔽₂

_≈ᵇ_ : 𝔽₂ → 𝔽₂ → 𝔹
zero ≈ᵇ one = false
one ≈ᵇ zero = false
_ ≈ᵇ _ = true

_+_ : 𝔽₂ → 𝔽₂ → 𝔽₂
zero + zero = zero
zero + one = one
one + zero = one
one + one = zero

_*_ : 𝔽₂ → 𝔽₂ → 𝔽₂
zero * zero = zero
zero * one = zero
one * zero = zero
one * one = one

-_ : 𝔽₂ → 𝔽₂
- x = x

_⁻¹ : 𝔽₂ → 𝔽₂
x ⁻¹ = x

+-comm : Commutative _≡_ _+_
+-comm zero zero = refl
+-comm zero one = refl
+-comm one zero = refl
+-comm one one = refl

*-comm : Commutative _≡_ _*_
*-comm zero zero = refl
*-comm zero one = refl
*-comm one zero = refl
*-comm one one = refl

+-idl : LeftIdentity _≡_ zero _+_
+-idl zero = refl
+-idl one = refl

+-idr : RightIdentity _≡_ zero _+_
+-idr zero = refl
+-idr one = refl

+-id : Identity _≡_ zero _+_
+-id = +-idl , +-idr

*-idl : LeftIdentity _≡_ one _*_
*-idl zero = refl
*-idl one = refl

*-idr : RightIdentity _≡_ one _*_
*-idr zero = refl
*-idr one = refl

*-id : Identity _≡_ one _*_
*-id = *-idl , *-idr

--not a nice way to do it
+-assoc : Associative _≡_ _+_
+-assoc zero zero zero = refl
+-assoc zero zero one = refl
+-assoc zero one zero = refl
+-assoc zero one one = refl
+-assoc one zero zero = refl
+-assoc one zero one = refl
+-assoc one one zero = refl
+-assoc one one one = refl

+-cong : Congruent₂ _≡_ _+_
+-cong refl refl = refl


+-isMagma : IsMagma _≡_ _+_
+-isMagma = record { isEquivalence = isEquivalence ;
                     ∙-cong = +-cong }

+-isSemigroup : IsSemigroup _≡_ _+_
+-isSemigroup = record { isMagma = +-isMagma ;
                         assoc = +-assoc }

+-isMonoid : IsMonoid _≡_ _+_ zero
+-isMonoid = record { isSemigroup = +-isSemigroup ;
                      identity = +-id }


*-cong : Congruent₂ _≡_ _*_
*-cong refl refl = refl

*-isMagma : IsMagma _≡_ _*_
*-isMagma = record { isEquivalence = isEquivalence ;
                     ∙-cong = *-cong }

*-assoc : Associative _≡_ _*_
*-assoc zero zero zero = refl
*-assoc zero zero one = refl
*-assoc zero one zero = refl
*-assoc zero one one = refl
*-assoc one zero zero = refl
*-assoc one zero one = refl
*-assoc one one zero = refl
*-assoc one one one = refl

*-isSemigroup : IsSemigroup _≡_ _*_
*-isSemigroup = record { isMagma = *-isMagma ;
                         assoc = *-assoc }  

*-isMonoid : IsMonoid _≡_ _*_ one
*-isMonoid = record { isSemigroup = *-isSemigroup
                    ; identity = *-id }

*-dist-+l : (_≡_ DistributesOverˡ _*_) _+_
*-dist-+l zero zero zero = refl
*-dist-+l zero zero one = refl
*-dist-+l zero one zero = refl
*-dist-+l zero one one = refl
*-dist-+l one zero zero = refl
*-dist-+l one zero one = refl
*-dist-+l one one zero = refl
*-dist-+l one one one = refl

*-dist-+r : (_≡_ DistributesOverʳ _*_) _+_
*-dist-+r zero zero zero = refl
*-dist-+r zero zero one = refl
*-dist-+r zero one zero = refl
*-dist-+r zero one one = refl
*-dist-+r one zero zero = refl
*-dist-+r one zero one = refl
*-dist-+r one one zero = refl
*-dist-+r one one one = refl

*-dist-+ : (_≡_ DistributesOver _*_) _+_
*-dist-+ = *-dist-+l , *-dist-+r

zero0*l : LeftZero _≡_ zero _*_
zero0*l zero = refl
zero0*l one = refl

zero0*r : RightZero _≡_ zero _*_
zero0*r zero = refl
zero0*r one = refl

zero0* : Zero _≡_ zero _*_
zero0* = zero0*l , zero0*r

-inverseₗ : LeftInverse _≡_ zero -_ _+_
-inverseₗ zero = refl
-inverseₗ one = refl

-inverseᵣ : RightInverse _≡_ zero -_ _+_
-inverseᵣ zero = refl
-inverseᵣ one = refl

-inverse : Inverse _≡_ zero -_  _+_
-inverse = -inverseₗ , -inverseᵣ

-cong : Congruent₁ _≡_ -_
-cong refl = refl

isGroup : IsGroup _≡_ _+_ zero -_
isGroup = record { isMonoid = +-isMonoid ;
                   inverse = -inverse ;
                   ⁻¹-cong = -cong }

+-isAbelianGroup : IsAbelianGroup _≡_ _+_ zero -_
+-isAbelianGroup = record {isGroup = isGroup ;
                           comm = +-comm }

isRing : IsRing _≡_ _+_ _*_ -_ zero one
isRing = record
           { +-isAbelianGroup = +-isAbelianGroup ;
             *-isMonoid = *-isMonoid ;
             distrib = *-dist-+ ;
             zero = zero0* }

isCommutativeRing : IsCommutativeRing _≡_ _+_ _*_ -_ zero one
isCommutativeRing = record { isRing = isRing ; *-comm = *-comm }


⁻¹inverse*ₗ : LeftInverse-¬0 _≡_ zero one _⁻¹ _*_
⁻¹inverse*ₗ zero p = ⊥-elim (p refl)
⁻¹inverse*ₗ one p = refl

⁻¹inverse*ᵣ : RightInverse-¬0 _≡_ zero one _⁻¹ _*_
⁻¹inverse*ᵣ zero p = ⊥-elim (p refl)
⁻¹inverse*ᵣ one p = refl

⁻¹inverse* : Inverse-¬0 _≡_ zero one _⁻¹ _*_
⁻¹inverse* = ⁻¹inverse*ₗ , ⁻¹inverse*ᵣ

≈ᵇ≈ : (x y : 𝔽₂) → (x ≈ᵇ y) ≡ true → x ≡ y
≈ᵇ≈ zero zero p = refl
≈ᵇ≈ one one p = refl

≈≈ᵇ : (x y : 𝔽₂) →  x ≡ y → (x ≈ᵇ y) ≡ true
≈≈ᵇ zero .zero refl = refl
≈≈ᵇ one .one refl = refl

¬≈ᵇ≈ : (a b : 𝔽₂) (p : (a ≈ᵇ b) ≡ false) (p₂ : a ≡ b) → ⊥
¬≈ᵇ≈ zero zero () p₂
¬≈ᵇ≈ zero one p ()
¬≈ᵇ≈ one zero p ()
¬≈ᵇ≈ one one () p₂

¬≈≈ᵇ : ∀ a b → ¬(a ≡ b) → (a ≈ᵇ b) ≡ false
¬≈≈ᵇ zero zero p = ⊥-elim (p refl)
¬≈≈ᵇ zero one p = refl
¬≈≈ᵇ one zero p = refl
¬≈≈ᵇ one one p = ⊥-elim (p refl)

gf2Field : Field 0ℓ 0ℓ
gf2Field = record
        { Carrier = 𝔽₂
        ; _≈_ = _≡_
        ; _≈ᵇ_ = _≈ᵇ_
        ; _+_ = _+_
        ; _*_ = _*_
        ; -_ = -_
        ; _⁻¹ = _⁻¹
        ; 0# = zero
        ; 1# = one
        ; isField = record
                      { isCommutativeRing = isCommutativeRing
                      ; ⁻¹inverse* = ⁻¹inverse*
                      ; ≈ᵇ≈ = ≈ᵇ≈
                      ; ¬≈ᵇ≈ = ¬≈ᵇ≈
                      ; ≈≈ᵇ = ≈≈ᵇ
                      ; ¬≈≈ᵇ = ¬≈≈ᵇ
                      }
        }

gf2Semiring : Semiring 0ℓ 0ℓ
gf2Semiring = record
                { Carrier = 𝔽₂
                ; _≈_ = _≡_
                ; _+_ = _+_
                ; _*_ = _*_
                ; 0# = zero
                ; 1# = one
                ; isSemiring = record {
                  isSemiringWithoutAnnihilatingZero = record {
                    +-isCommutativeMonoid = record {
                      isMonoid = +-isMonoid ;
                      comm = +-comm } ;
                    *-isMonoid = *-isMonoid ;
                    distrib = *-dist-+ } ;
                  zero = zero0* }
                }
