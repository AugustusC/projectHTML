open import Level using (Level)
open import Data.Nat as N
open import Relation.Binary
open import Data.Fin

module naive where

open import Data.Vec hiding (sum) renaming (Vec to Vecₙ) public

variable
  a' : Level
  A' : Set a'

Matₙ : (A : Set a') → (n m : ℕ) → Set a'
Matₙ A m n = Vecₙ (Vecₙ A n) m

lookupₘ : ∀ {n m} → Matₙ A' n m → Fin n → Fin m → A'
lookupₘ (v ∷ vs) zero m = lookup v m
lookupₘ (v ∷ vs) (suc n) m = lookupₘ vs n m
