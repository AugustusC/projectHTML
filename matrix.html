--Lift relationships from values to vectors of values
Pw : REL A B ℓ → REL (Vector A n) (Vector B n) ℓ
Pw R u v = ∀ i → R (u i) (v i)

  _+ᵥ_ : (t v : Vector Carrier n) → Vector Carrier n

  _+ₘ_ : (M N : Matrix Carrier m n) → Matrix Carrier m n
  (M +ₘ N) i j = (M i +ᵥ N i) j


  +ₘ-identityˡ : (M : Matrix Carrier m n) → 0ₘ +ₘ M ≈ₘ M
  +ₘ-identityˡ M i j = +-identityˡ (M i j)

  +ₘ-identityʳ : (M : Matrix Carrier m n) → M +ₘ 0ₘ ≈ₘ M
  +ₘ-identityʳ M i j = +-identityʳ (M i j)

  _==_ : (i j : Fin n) → Bool
  zero == zero = true
  zero == suc j = false
  suc i == zero = false

  1ₘ : Matrix Carrier m m
  1ₘ i j = if i == j then 1# else 0#

  sum : Vector Carrier n → Carrier
  sum {zero} v = 0#
  sum {suc n} v = v zero + sum {n} (v ∘ suc)

  _*ₘ_ : Matrix Carrier m n → Matrix Carrier n o → Matrix Carrier m o
  (M *ₘ N) i k = sum λ j → M i j * N j k

  _*ₛ_ : Carrier → Matrix Carrier m n → Matrix Carrier m n
  (s *ₛ M) i j = s * (M i j)



  *ₛ-assoc : (s t : Carrier) → (M : Matrix Carrier m n) → (s * t) *ₛ M ≈ₘ s *ₛ (t *ₛ M)
  *ₛ-assoc s t M i j = *-assoc s t (M i j)

  *ₛ-identity : (M : Matrix Carrier m n) → 1# *ₛ M ≈ₘ M
  *ₛ-identity M i j = *-identityˡ (M i j)

  *ₛ-zero : (M : Matrix Carrier m n) → 0# *ₛ M ≈ₘ 0ₘ
  *ₛ-zero M i j = zeroˡ (M i j)

  *ₛ-zeroₘ : (s : Carrier) → s *ₛ 0ₘ {m} {m} ≈ₘ 0ₘ 
  *ₛ-zeroₘ s i j = zeroʳ s

  --Scalar can be distributed over matrix addition
  *ₛ+ₘdistrib : (s : Carrier) → (M N : Matrix Carrier m n)
    → s *ₛ (M +ₘ N) ≈ₘ (s *ₛ M) +ₘ (s *ₛ N)
  *ₛ+ₘdistrib s M N i j = distribˡ s (M i j) (N i j)

  --Matrix can be distributed over scalar addition
  m*ₛdistrib : (s t : Carrier) → (M : Matrix Carrier m n)
    → (s + t) *ₛ M ≈ₘ (s *ₛ M) +ₘ (t *ₛ M)
  m*ₛdistrib s t M i j = distribʳ (M i j) s t
 
  sum-cong : {u v : Vector _ n} → u ≈ᵥ v → sum u ≈ sum v
  sum-cong {zero} q = refl
  sum-cong {suc n} q = +-cong (q zero) (sum-cong (λ i → q (suc i)))

  +-sum : (u v : Vector _ n) → sum (u +ᵥ v) ≈ (sum u) + (sum v)
  +-sum {zero} u v = sym (+-identityʳ _)
  +-sum {suc n} u v = 
    begin
      sum (u +ᵥ v) ≡⟨⟩
      (u +ᵥ v) zero + sum {n} ((u ∘ suc) +ᵥ (v ∘ suc))
        ≈⟨ +-cong refl (+-sum (u ∘ suc) _) ⟩
      (u +ᵥ v) zero + (sum {n} (u ∘ suc) + sum {n} (v ∘ suc)) ≡⟨⟩
      (u zero + v zero) + (sum {n} (u ∘ suc) + sum {n} (v ∘ suc)) ≈⟨ arr (u zero) _ _ _ ⟩
      (u zero + sum {n} (u ∘ suc)) + (v zero + sum {n} (v ∘ suc)) ≡⟨⟩
      (sum u) + (sum v)
    ∎ where arr : ∀ a b c d → (a + b) + (c + d) ≈ (a + c) + (b + d)
            arr a b c d =
              begin
                (a + b) + (c + d) ≈⟨ sym (+-assoc (a + b) c d) ⟩
                ((a + b) + c) + d ≈⟨ +-cong (+-assoc a b c) refl ⟩
                (a + (b + c)) + d ≈⟨ +-cong (+-cong refl (+-comm b c)) refl ⟩
                (a + (c + b)) + d ≈⟨ +-cong (sym (+-assoc a c b)) refl ⟩
                ((a + c) + b) + d ≈⟨ +-assoc (a + c) b d ⟩
                (a + c) + (b + d)
              ∎

  *-sumˡ : ∀ x (u : Vector _ n) → (sum λ i → x * u i) ≈ x * sum u
  *-sumˡ {zero} x u = sym (zeroʳ x)
  *-sumˡ {suc n} x u = trans (+-cong refl (*-sumˡ {n} x (u ∘ suc)))
                             (sym (distribˡ x (u zero) (sum (u ∘ suc))))

  *-sumʳ : ∀ x (u : Vector _ n) → (sum λ i → u i * x) ≈ sum u * x
  *-sumʳ {zero} x u = sym (zeroˡ x)
  *-sumʳ {suc n} x u = trans (+-cong refl (*-sumʳ {n} x (u ∘ suc)))
                             (sym (distribʳ x (u zero) (sum (λ z → u (suc z)))))
  
  *ₘ-identityˡ : (M : Matrix Carrier m n) → 1ₘ *ₘ M ≈ₘ M
  *ₘ-identityˡ {suc m} {n} M zero k =
    begin
      (1ₘ *ₘ M) zero k                               ≡⟨⟩
      1ₘ {suc m} zero zero * M zero k
        + (sum λ j → 1ₘ zero (suc j) * M (suc j) k ) ≡⟨⟩
      1# * M zero k + (sum λ j → 0# * M (suc j) k)
           ≈⟨ +-cong (*-identityˡ _) (*-sumˡ 0# (λ j → M (suc j) k)) ⟩
      M zero k + 0# * (sum λ j → M (suc j) k)        ≈⟨ +-cong refl (zeroˡ _) ⟩
      M zero k + 0#                                  ≈⟨ +-identityʳ (M zero k) ⟩
      M zero k
    ∎
  *ₘ-identityˡ {suc m} {n} M (suc i) k =
    begin
     (1ₘ *ₘ M) (suc i) k                               ≡⟨⟩
     (sum λ j → 1ₘ (suc i) j * M j k)                  ≡⟨⟩
     1ₘ (suc i) zero * M zero k
       + (sum λ j → 1ₘ (suc i) (suc j) * M (suc j) k)  ≡⟨⟩ --? Using property of 1ₘ that suc i suc j is equal to i j
     0# * M zero k + (sum λ j → 1ₘ i j * M (suc j) k)    ≡⟨⟩
     0# * M zero k + (1ₘ *ₘ (λ j k → M (suc j) k)) i k
       ≈⟨ +-cong (zeroˡ _) (*ₘ-identityˡ (λ j k → M (suc j) k) i k) ⟩
     0# + M (suc i) k                                  ≈⟨ +-identityˡ _ ⟩
     M (suc i) k
    ∎

  *ₘ-identityʳ : (M : Matrix Carrier m n) → M *ₘ 1ₘ ≈ₘ M
  *ₘ-identityʳ {suc m} {suc n} M i zero =
    begin
      (M *ₘ 1ₘ) i zero                               ≡⟨⟩
      (sum λ j → M i j * 1ₘ j zero)                  ≡⟨⟩
      (M i zero * 1ₘ {suc m} zero zero)
        + (sum λ j → M i (suc j) * 1ₘ (suc j) zero)  ≡⟨⟩
      (M i zero * 1#) + (sum λ j → M i (suc j) * 0#)
                      ≈⟨ +-cong (*-identityʳ _) (*-sumʳ 0# λ j → M i (suc j)) ⟩
      M i zero + (sum λ j → M i (suc j)) * 0#        ≈⟨ +-cong refl (zeroʳ _) ⟩
      M i zero + 0#                                  ≈⟨ +-identityʳ _ ⟩ 
      M i zero 
    ∎
  *ₘ-identityʳ {suc m} {suc n} M i (suc k) =
    begin
      (M *ₘ 1ₘ) i (suc k)              ≡⟨⟩
      (sum λ j → M i j * 1ₘ j (suc k)) ≡⟨⟩
      (M i zero * 1ₘ zero (suc k))
        + (sum λ j → M i (suc j) * 1ₘ (suc j) (suc k)) ≡⟨⟩
      (M i zero * 0#)
        + (sum λ j → M i (suc j) * 1ₘ j k)   ≈⟨ +-cong (zeroʳ _) refl ⟩
      0# + ((λ i j → M i (suc j)) *ₘ 1ₘ) i k ≈⟨ +-cong refl (*ₘ-identityʳ (λ i j → M i (suc j)) i k) ⟩
      0# + M i (suc k)                       ≈⟨ +-identityˡ _ ⟩ 
      M i (suc k)
    ∎

  *ₘ-zeroˡ : (M : Matrix Carrier m n) → 0ₘ {m} *ₘ M ≈ₘ 0ₘ
  *ₘ-zeroˡ M i k =
    begin
      (0ₘ *ₘ M) i k              ≡⟨⟩
      (sum λ j → 0ₘ i j * M j k) ≡⟨⟩
      (sum λ j → 0# * M j k)     ≈⟨ *-sumˡ 0# (λ j → M j k) ⟩
      0# * (sum λ j → M j k)     ≈⟨ zeroˡ _ ⟩ 
      0#
    ∎

  *ₘ-zeroʳ : (M : Matrix Carrier m n) → M *ₘ 0ₘ ≈ₘ 0ₘ {m} {n}
  *ₘ-zeroʳ M i k =
    begin
      (M *ₘ 0ₘ) i k              ≡⟨⟩
      (sum λ j → M i j * 0ₘ j k) ≡⟨⟩
      (sum λ j → M i j * 0#)     ≈⟨ *-sumʳ 0# (λ j → M i j) ⟩
      (sum λ j → M i j) * 0#     ≈⟨ zeroʳ _ ⟩
      0#
    ∎

  pdistₗ : (u v w : Vector Carrier n)
    → (sum λ i → u i * (v i + w i)) ≈ (sum λ i → (u i * v i) + (u i * w i))
  pdistₗ {zero} u v w = refl
  pdistₗ {suc n} u v w = +-cong (distribˡ (u zero) _ _) (pdistₗ (u ∘ suc) _ _)

  *ₘ+ₘdistribₗ :
      (M : Matrix Carrier m n)
    → (N O : Matrix Carrier n o)
    ----------------------------
    → M *ₘ (N +ₘ O) ≈ₘ (M *ₘ N) +ₘ (M *ₘ O)

  *ₘ+ₘdistribₗ M N O i j =
    begin
      (M *ₘ (N +ₘ O)) i j ≡⟨⟩
      (sum λ k → M i k * (N k j + O k j))
                          ≈⟨ pdistₗ (λ k → M i k) (λ k → N k j) (λ k → O k j) ⟩
      (sum λ k → (M i k * N k j) + (M i k * O k j))
                          ≈⟨ +-sum (λ k → (M i k * N k j)) (λ k → (M i k * O k j)) ⟩ 
      ((M *ₘ N) +ₘ (M *ₘ O)) i j
    ∎ 

  pdistᵣ :
    (u v w : Vector Carrier n)
    → (sum λ i → (u i + v i) * w i) ≈ (sum λ i → (u i * w i) + (v i * w i))
  pdistᵣ {zero} u v w = refl
  pdistᵣ {suc n} u v w = +-cong (distribʳ (w zero) _ _) (pdistᵣ (u ∘ suc) _ _)

  *ₘ+ₘdistribᵣ :
      (M N : Matrix Carrier m n)
    → (O : Matrix Carrier n o)
    ----------------------------
    → (M +ₘ N) *ₘ O ≈ₘ (M *ₘ O) +ₘ (N *ₘ O)

  *ₘ+ₘdistribᵣ M N O i j =
    begin
      ((M +ₘ N) *ₘ O) i j ≡⟨⟩
      (sum λ k → (M i k + N i k) * O k j)
                          ≈⟨ pdistᵣ (λ k → M i k) _ _ ⟩
      (sum λ k → (M i k * O k j) + (N i k * O k j))
                          ≈⟨ +-sum (λ k → (M i k * O k j)) _ ⟩
      ((M *ₘ O) +ₘ (N *ₘ O)) i j
    ∎

  pdisₗ : (u v : Vector Carrier n) → (s : Carrier)
    → (sum λ i → (s * u i) * v i) ≈ (sum λ i → s * (u i * v i))
  pdisₗ {zero} u v s = refl
  pdisₗ {suc n} u v s =
    +-cong (*-assoc s (u zero) (v zero)) (pdisₗ (λ z → u (suc z)) _ s)

  -- M *ₘ (s *ₛ N) ≈ₘ s *ₛ (M *ₘ N)

  *ₘ*ₛdistribₗ :
      (M : Matrix Carrier m n)
    → (N : Matrix Carrier n o)
    → (s : Carrier)
    --------------------------
    → (s *ₛ M) *ₘ N ≈ₘ s *ₛ(M *ₘ N)
  *ₘ*ₛdistribₗ M N s i j =
    begin
      ((s *ₛ M) *ₘ N) i j             ≡⟨⟩
      (sum λ k → (s * M i k) * N k j) ≈⟨ pdisₗ (M i) _ s ⟩
      (sum λ k → s * (M i k * N k j)) ≈⟨ *-sumˡ s (λ k → (M i k * N k j)) ⟩
      s * (sum λ k → M i k * N k j)   ≡⟨⟩
      (s *ₛ(M *ₘ N)) i j
    ∎


--Cant prove without knowing that multiplication is commutative
  pdisᵣ :
      (*-comm : Commutative _≈_ _*_)
    → (u v : Vector Carrier n)
    → (s : Carrier)
    --------------------------------
    → (sum λ i → u i * (s * v i)) ≈ (sum λ i → s * (u i * v i))
  pdisᵣ {zero} *-comm u v s = refl
  pdisᵣ {suc n} *-comm u v s = +-cong (arr (u zero) _ _) (pdisᵣ *-comm (λ z → u (suc z)) _ _) where
    arr : ∀ a b c → a * (b * c) ≈ b * (a * c)
    arr a b c =
      begin
        (a * (b * c)) ≈⟨ sym (*-assoc a b c) ⟩
        (a * b) * c   ≈⟨ *-cong (*-comm a b) refl ⟩
        (b * a) * c   ≈⟨ *-assoc b a c ⟩
        (b * (a * c))
      ∎

  *ₘ*ₛdistribᵣ :
      (*-comm : Commutative _≈_ _*_)
    → (M : Matrix Carrier m n)
    → (N : Matrix Carrier n o)
    → (s : Carrier)
    --------------------------------
    → M *ₘ (s *ₛ N) ≈ₘ s *ₛ (M *ₘ N)
  *ₘ*ₛdistribᵣ *-comm M N s i j =
    begin
      (M *ₘ (s *ₛ N)) i j             ≡⟨⟩
      (sum λ k → M i k * (s * N k j)) ≈⟨ pdisᵣ *-comm (M i) _ _ ⟩
      (sum λ k → s * (M i k * N k j)) ≈⟨ *-sumˡ s (λ k → M i k * N k j) ⟩
      s * (sum λ k → M i k * N k j)   ≡⟨⟩
      (s *ₛ (M *ₘ N)) i j
    ∎
