open import Level
open import Relation.Binary
open import Algebra

open import prop

module struct where

open import Data.Bool renaming (Bool to 𝔹)

record Field c ℓ : Set (suc (c ⊔ ℓ)) where
  infix 8 -_
  infix 8 _⁻¹
  infixl 7 _*_
  infixl 6 _+_
  infix 4 _≈_

  field
    Carrier : Set c
    _≈_ : Rel Carrier ℓ
    _≈ᵇ_ : Carrier → Carrier → 𝔹
    _+_ : Op₂ Carrier
    _*_ : Op₂ Carrier
    -_  : Op₁ Carrier
    _⁻¹ : Op₁ Carrier
    0#  : Carrier
    1#  : Carrier
    isField : IsField _≈_ _≈ᵇ_ _+_ _*_ -_ _⁻¹ 0# 1#

  open IsField isField public
